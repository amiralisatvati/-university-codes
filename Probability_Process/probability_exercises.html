Exercise: Random Variables and Distributions

1. Discrete Probability Distributions

# Binomial distribution
n <- 10; p <- 0.3
x <- 0:n
prob <- dbinom(x, n, p)
plot(x, prob, type="h", lwd=2, col="blue")

2. Continuous Distributions

# Normal distribution
x <- seq(-4, 4, 0.01)
y <- dnorm(x)
plot(x, y, type="l", col="red", lwd=2)
